<script>

let seconds = 0;

let timerInterval = null;

let completed = {};

 

window.onload = displayLeaderboard;

 

function tStart() {

if (timerInterval) return;

if (!document.getElementById('uname').value.trim()) {

showToast("Enter your name first!");

return;

}

document.getElementById('timer').classList.add('on');

document.getElementById('startB').disabled = true;

document.getElementById('stopB').disabled = false;

timerInterval = setInterval(() => {

seconds++;

const h = Math.floor(seconds / 3600).toString().padStart(2, '0');

const m = Math.floor((seconds % 3600) / 60).toString().padStart(2, '0');

const s = (seconds % 60).toString().padStart(2, '0');

document.getElementById('timer').textContent = `${h}:${m}:${s}`;

}, 1000);

}

 

function tStop() {

if (Object.keys(completed).length < 3) {

if(!confirm("Finish without finding everything?")) return;

}

clearInterval(timerInterval);

timerInterval = null;

document.getElementById('timer').classList.remove('on');

saveScore(document.getElementById('uname').value, seconds);

displayLeaderboard();

showToast("Score Saved!");

}

 

function saveScore(name, time) {

let scores = JSON.parse(localStorage.getItem('huntScores') || '[]');

scores.push({ name, time });

scores.sort((a, b) => a.time - b.time);

localStorage.setItem('huntScores', JSON.stringify(scores.slice(0, 5)));

}

 

function displayLeaderboard() {

const scores = JSON.parse(localStorage.getItem('huntScores') || '[]');

if (scores.length === 0) return;

document.getElementById('lbCard').classList.add('active');

document.getElementById('lbContent').innerHTML = scores.map((s, i) => `

<div class="lb-row ${i === 0 ? 'gold' : ''}">

<span>${i + 1}. ${s.name}</span>

<span>${Math.floor(s.time/60)}m ${s.time%60}s</span>

</div>

`).join('');

}

 

function exportLeaderboard() {

const scores = JSON.parse(localStorage.getItem('huntScores') || '[]');

const canvas = document.getElementById('exportCanvas');

const ctx = canvas.getContext('2d');

canvas.width = 400;

canvas.height = 300 + (scores.length * 40);

// Background

ctx.fillStyle = '#1e293b';

ctx.fillRect(0, 0, canvas.width, canvas.height);

// Header

ctx.fillStyle = '#f59e0b';

ctx.font = 'bold 24px Arial';

ctx.fillText('🏆 Urban Quest Top 5', 40, 50);

// Rows

ctx.font = '18px Arial';

scores.forEach((s, i) => {

ctx.fillStyle = (i === 0) ? '#ffd700' : '#f8fafc';

const text = `${i + 1}. ${s.name}`;

const time = `${Math.floor(s.time/60)}m ${s.time%60}s`;

ctx.fillText(text, 40, 100 + (i * 45));

ctx.fillText(time, 300, 100 + (i * 45));

ctx.strokeStyle = '#334155';

ctx.beginPath(); ctx.moveTo(40, 110 + (i * 45)); ctx.lineTo(360, 110 + (i * 45)); ctx.stroke();

});

 

// Branding

ctx.fillStyle = '#6366f1';

ctx.font = 'italic 14px Arial';

ctx.fillText('Generated by Urban Quest App', 110, canvas.height - 20);

 

const link = document.createElement('a');

link.download = 'Leaderboard.png';

link.href = canvas.toDataURL();

link.click();

showToast("Leaderboard Image Saved!");

}

 

function tReset() {

if(confirm("Clear everything?")) { localStorage.removeItem('huntScores'); location.reload(); }

}

 

function handleFile(input, id) {

if (input.files && input.files[0]) {

const img = document.getElementById(`pimg${id}`);

img.src = URL.createObjectURL(input.files[0]);

img.style.display = 'block';

document.getElementById(`item${id}`).classList.add('done');

document.getElementById(`chk${id}`).innerHTML = '✓';

document.getElementById(`chk${id}`).style.background = 'var(--success)';

completed[id] = true;

updateProgress();

}

}

 

function updateProgress() {

const count = Object.keys(completed).length;

document.getElementById('pbar').style.width = (count / 3 * 100) + '%';

document.getElementById('prog').textContent = `${count}/3`;

}

 

function shareResults() {

const name = document.getElementById('uname').value || "Explorer";

const time = document.getElementById('timer').textContent;

const text = `🎯 *Urban Quest Update!*\n👤 Name: ${name}\n⏱ Time: ${time}\nItems: ${Object.keys(completed).length}/3`;

window.open(`https://wa.me/?text=${encodeURIComponent(text)}`);

}

 

function showToast(msg) {

const t = document.getElementById('toast');

t.textContent = msg;

t.style.display = 'block';

setTimeout(() => t.style.display = 'none', 2000);

}
